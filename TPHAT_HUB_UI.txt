local function playSoundInstant(soundInstance)
	if soundInstance and soundInstance.SoundId ~= "" then
		task.spawn(function()

local function applyGlobalDrag(triggerInstance)
	local draggingHub, dragInputHub, dragStartHub, startPosHub
	
	triggerInstance.InputBegan:Connect(function(input)
		if (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1) then
			local mousePos = UIS:GetMouseLocation()
			local handlePos = MainFrame.AbsolutePosition + MainFrame.AbsoluteSize - Vector2.new(25, 25)
			if mousePos.X >= handlePos.X and mousePos.Y >= handlePos.Y then
				return
			end
			
			draggingHub = true
			dragStartHub = input.Position
			startPosHub = MainFrame.Position
			
			if triggerInstance == MoveHandle then
				TweenService:Create(MoveHandle, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					Size = UDim2.new(0.2, 0, 0, 8),
					BackgroundColor3 = AccentColor
				}):Play()
			end
			
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then 
					draggingHub = false 
					if triggerInstance == MoveHandle then
						TweenService:Create(MoveHandle, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
							Size = UDim2.new(0.2, 0, 0, 4),
							BackgroundColor3 = Color3.fromRGB(80, 85, 100)
						}):Play()
					end
				end
			end)
		end
	end)
	
	triggerInstance.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement then 
			dragInputHub = input 
		end
	end)
	
	UIS.InputChanged:Connect(function(input)
		if input == dragInputHub and draggingHub then
			local delta = input.Position - dragStartHub
			MainFrame.Position = UDim2.new(startPosHub.X.Scale, startPosHub.X.Offset + delta.X, startPosHub.Y.Scale, startPosHub.Y.Offset + delta.Y)
		end
	end)
end

applyGlobalDrag(MoveHandle)
applyGlobalDrag(TopBar)
applyGlobalDrag(Sidebar)
applyGlobalDrag(CenterArea)

local IsMinimized = false

local function toggleMenu(forceState)
	local targetState = forceState
	if targetState == nil then
		targetState = not MainFrame.Visible
	end
	
	if targetState == true then
		MainFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
		MainFrame.Visible = true
		
		if IsMinimized then
			MainFrame.Size = UDim2.new(0, BaseWidth, 0, 0)
			local openTween = TweenService:Create(MainFrame, TweenInfo.new(0.4, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
				Size = UDim2.new(0, BaseWidth, 0, 45)
			})
			openTween:Play()
		else
			MainFrame.Size = UDim2.new(0, BaseWidth, 0, 0)
			local openTween = TweenService:Create(MainFrame, TweenInfo.new(0.4, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
				Size = UDim2.new(0, BaseWidth, 0, BaseHeight)
			})
			openTween:Play()
			Container.Size = UDim2.new(1, 0, 1, -45)
		end
	else
		local closeTween = TweenService:Create(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
			Size = UDim2.new(0, BaseWidth, 0, 0)

local dragging
local dragInput
local dragStart
local startPos

local function update(input)
	local delta = input.Position - dragStart
	ToggleButton.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

ToggleButton.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = true
		dragStart = input.Position
		startPos = ToggleButton.Position

		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

ToggleButton.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement then
		dragInput = input
	end
end)

UIS.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		update(input)
	end
end)
